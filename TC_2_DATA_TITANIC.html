import pandas as pd
# Load the Titanic dataset (assuming it's already loaded into 'df')
df = pd.read_csv('Titanic-Dataset.csv')
# Convert 'Sex' column using Label Encoding (Male = 0, Female = 1)
df['Sex'] = df['Sex'].map({'male': 0, 'female': 1})
# Convert 'Embarked' column using Label Encoding (C = 0, Q = 1, S = 2)
df['Embarked'] = df['Embarked'].map({'C': 0, 'Q': 1, 'S': 2})
# Optionally, use One-Hot Encoding for other categorical columns
